\makeatletter
% Environment 0itemize behaves like itemize but with \topsep 0pt and \partopsep 0pt.
\newenvironment{0itemize}{%
  \topsep=\z@
  \partopsep=\z@
  \def\@listi{\leftmargin\leftmargini \parsep\z@ \topsep\z@ \itemsep\z@}
  \let\@listI\@listi
  \itemize
}{%
  \enditemize
}
\makeatother	% back to normal usage of @
